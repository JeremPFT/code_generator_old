with Ada.Text_IO;

with Model.Comment.Tests;
with Model.Array_Def;
with Model.Class_Def.Tests;
with Model.Enum_Def;
with Model.Subprogram;

procedure Model.Tests
is
  package T_IO renames Ada.Text_IO;

  Obj_Comment_1 : access Comment.Object_T := null;

begin
  T_IO.Put_Line ("debut des tests du model");

  Obj_Comment_1 := Comment.Create ("un commentaire");
  Obj_Comment_1.Add_Comment ("commentaire a");
  Obj_Comment_1.Add_Comment ("commentaire b");
  Obj_Comment_1.Add_Comment ("commentaire c");

  T_IO.Put_Line ("commentaire : " & Obj_Comment_1.Get_Body);
  T_IO.Put_Line ("# commentaire : " & Obj_Comment_1.Number_Of_Comments'Img);

  for I in 1 .. Obj_Comment_1.Number_Of_Comments loop
    T_IO.Put_Line ("commentaire : " & I'Img & " : " &
                     Obj_Comment_1.Get_Comment (I));
  end loop;

end Model.Tests;
