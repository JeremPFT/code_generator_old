with Ada.Containers.Vectors;

with Model.Element;

package Model.Comment is

  type Object_T is new Model.Element.Object_T with private;

  type Reference_T is access all Object_T;

  type Class_T is access all Object_T'Class;

  package Vectors is new Ada.Containers.Vectors
    (Element_Type => Class_T,
     Index_Type   => Positive);

  subtype Vector_T is Vectors.Vector;

  function Create
    (Text : in String)
    return Class_T;

  function Get_Body
    (Self : in Object_T)
    return String;

private

  type Object_T is new Element.Object_T
    with record
      Comment_Body : access String := null;
    end record;

  function Get_Body
    (Self : in Object_T)
    return String
    is (Self.Comment_Body.all);

end Model.Comment;
