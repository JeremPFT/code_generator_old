with Ada.Text_IO;

with Test_Utils;

with Reflection;

package body Model.Class_Def.Tests is

  package Unit is new Reflection;

  package T_IO renames Ada.Text_IO;

  procedure Empty is
  begin
    null;
  end Empty;

  Obj_Package_1 : access Package_Def.Object_T := null;
  Obj_Class_1   : access Object_T             := null;

  procedure Set_Up;

  procedure Run;

  procedure Set_Up
  is
  begin
    Obj_Package_1 := Package_Def.Create ("package_a");

    Obj_Class_1 := Create (Owner_Package => Obj_Package_1,
                           Name          => "class_a");
  end Set_Up;

  procedure Run
  is
    Number : Integer renames Obj_Class_1.Number_Of_Comments;
  begin
    Test_Utils.Section (Unit_Name => Unit.Name);

    T_IO.Put_Line ("class : " & Obj_Class_1.Get_Name);
    T_IO.Put_Line ("nbr comments : " & Number'Img);

    for I in 1 .. Number loop
      T_IO.Put_Line ("commentaire : " & I'Img & " : " &
                       Obj_Class_1.Get_Comment (I));
    end loop;

  end Run;

begin

  Set_Up;

  Run;

end Model.Class_Def.Tests;
