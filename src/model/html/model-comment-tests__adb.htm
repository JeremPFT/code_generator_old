with Ada.Text_IO;

with Test_Utils;

with Reflection;

package body Model.Comment.Tests is

  package Unit is new Reflection;

  package T_IO renames Ada.Text_IO;

  procedure Empty is
  begin
    null;
  end Empty;

  Obj_Comment_1 : access Object_T := null;

  procedure Set_Up;

  procedure Run;

  procedure Set_Up
  is
  begin
    Obj_Comment_1 := Create ("a comment");
    Obj_Comment_1.Add_Comment ("comment a");
    Obj_Comment_1.Add_Comment ("comment b");
    Obj_Comment_1.Add_Comment ("comment c");
  end Set_Up;

  procedure Run
  is
    Number : Integer renames Obj_Comment_1.Number_Of_Comments;
  begin
    Test_Utils.Section (Unit_Name => Unit.Name);

    T_IO.Put_Line ("comment : " & Obj_Comment_1.Get_Body);
    T_IO.Put_Line ("nbr comments : " & Number'Img);

    for I in 1 .. Number loop
      T_IO.Put_Line ("commentaire : " & I'Img & " : " &
                       Obj_Comment_1.Get_Comment (I));
    end loop;

  end Run;

begin

  Set_Up;

  Run;

end Model.Comment.Tests;
